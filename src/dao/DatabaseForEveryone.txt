    username VARCHAR(50) NOT NULL unique,

    ImageSrc VARCHAR(100) DEFAULT NULL,
    publisher VARCHAR(100),
    author VARCHAR(100) NOT NULL,
    isbn VARCHAR(50) UNIQUE,
